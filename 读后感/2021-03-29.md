可以使用syslog包，直接将日志信息往系统日志那边写

通过阅读log，可以进行自定义自己的日志，大致知道开源的log实现原理

一个简单的自定义日志：
```go
package main

import (
	"io"
	"log"
	"os"
)

type MyLog struct {
	Info *log.Logger
	Warning *log.Logger
	Error * log.Logger
}

var logs MyLog

func init(){
	errFile,err:=os.OpenFile("errors.log",os.O_CREATE|os.O_WRONLY|os.O_APPEND,0666)
	if err!=nil{
		log.Fatalln("打开日志文件失败：",err)
	}

	logs.Info = log.New(os.Stdout,"Info:",log.Ldate | log.Ltime | log.Lshortfile)
	logs.Warning = log.New(os.Stdout,"Warning:",log.Ldate | log.Ltime | log.Lshortfile)
	logs.Error = log.New(io.MultiWriter(os.Stderr,errFile),"Error:",log.Ldate | log.Ltime | log.Lshortfile)
}

func main() {
	logs.Info.Println("Info")
	logs.Warning.Printf("Warning")
	logs.Error.Println("Error")
}
```
