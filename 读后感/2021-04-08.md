func GetCurrentGoroutineInfo() {
	curg := getg()
	println("当前协程id：", curg.goid)
	println("当前协程绑定的mid：", curg.m.id)
	println("当前m执行的协程id：", curg.m.curg.goid)
}

组等待的小测试：
```go
package main

import (
	"fmt"
	"sync"
	"time"
)

func main() {
	var wg sync.WaitGroup
	a := 1 << 32
	wg.Add(a)

	for i := 0; i < 1; i++ {
		wg.Add(1)
		go func(id int) {
			for {
				time.Sleep(time.Second)
				fmt.Println("id:", id, ",", time.Now())
			}
		}(i)
		wg.Done()
	}
	wg.Wait()
}
```
